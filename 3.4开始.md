4.2.2 用户认证与权限控制实现
基于 JWT（JSON Web Token）实现用户认证，用户登录成功后生成 JWT 令牌，前端存储令牌并在后续请求中携带，后端拦截请求并校验令牌的有效性和权限信息。
JWT 工具类实现：

运行
package com.gym.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import java.util.Date;

@Component
public class JwtUtil {
    // 密钥
    @Value("${jwt.secret}")
    private String secret;
    // 过期时间（小时）
    @Value("${jwt.expire}")
    private int expire;

    // 生成JWT令牌
    public String generateToken(Integer userId, Integer role) {
        Date now = new Date();
        Date expireDate = new Date(now.getTime() + expire * 3600 * 1000L);
        return Jwts.builder()
                .setSubject(userId.toString())
                .claim("role", role)
                .setIssuedAt(now)
                .setExpiration(expireDate)
                .signWith(SignatureAlgorithm.HS256, secret)
                .compact();
    }

    // 解析JWT令牌，获取Claims
    public Claims parseToken(String token) {
        return Jwts.parser()
                .setSigningKey(secret)
                .parseClaimsJws(token)
                .getBody();
    }

    // 校验令牌是否过期
    public boolean isTokenExpired(String token) {
        Claims claims = parseToken(token);
        return claims.getExpiration().before(new Date());
    }
}

    @Autowired
    private UserService userService;
    @Autowired
    private JwtUtil jwtUtil;

    @PostMapping("/login")
    public Result<Map<String, Object>> login(@RequestBody User user) {
        // 验证账号密码
        User loginUser = userService.login(user.getUsername(), user.getPassword());
        if (loginUser == null) {
            return Result.error("账号或密码错误");
        }

        // 封装返回数据
        Map<String, Object> data = new HashMap<>();
        data.put("token", token);
        data.put("userInfo", loginUser);
        return Result.success(data);
    }

权限拦截器实现：
java
运行
package com.gym.config;

import com.gym.common.Result;
import com.gym.util.JwtUtil;
import com.fasterxml.jackson.databind.ObjectMapper;
import io.jsonwebtoken.Claims;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;
import org.springframework.web.servlet.HandlerInterceptor;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.PrintWriter;

@Component
public class AuthInterceptor implements HandlerInterceptor {
    @Autowired
    private JwtUtil jwtUtil;

    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        // 获取请求头中的令牌
        String token = request.getHeader("token");
        if (token == null || token.isEmpty()) {
            // 返回未授权响应
            response.setContentType("application/json;charset=utf-8");
            PrintWriter writer = response.getWriter();
            writer.write(new ObjectMapper().writeValueAsString(Result.error("未登录，请先登录")));
            writer.close();
            return false;
        }
        // 校验令牌是否过期
        if (jwtUtil.isTokenExpired(token)) {
            response.setContentType("application/json;charset=utf-8");
            PrintWriter writer = response.getWriter();
            writer.write(new ObjectMapper().writeValueAsString(Result.error("登录已过期，请重新登录")));
            writer.close();
            return false;
        }
        // 解析令牌，获取用户信息和权限
        Claims claims = jwtUtil.parseToken(token);
        Integer userId = Integer.parseInt(claims.getSubject());
        Integer role = claims.get("role", Integer.class);
        // 将用户信息存入请求域，便于后续接口使用
        request.setAttribute("userId", userId);
        request.setAttribute("role", role);
        return true;
    }
}

基于 Axios 封装请求拦截器和响应拦截器，请求拦截器添加令牌到请求头，响应拦截器统一处理异常：

运行
import axios from 'axios'
import { Message } from 'element-ui'

// 创建Axios实例
const service = axios.create({
  baseURL: 'http://localhost:8080',
  timeout: 5000
})

// 请求拦截器
service.interceptors.request.use(
  config => {
    // 添加令牌到请求头
    const token = localStorage.getItem('token')
    if (token) {
      config.headers['token'] = token
    }
    return config
  },
  error => {
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  response => {
    const res = response.data
    // 状态码非200则为异常
    if (res.code !== 200) {
      Message.error(res.msg || '请求失败')
      return Promise.reject(res)
    }
    return res
  },
  error => {
    Message.error(error.message || '服务器异常')
    return Promise.reject(error)
  }
)

export default service

4.3.4 权限路由控制实现
基于 Vue Router 实现路由拦截，根据用户角色过滤可访问的路由，未登录用户拦截到登录页：
javascript
运行
import Vue from 'vue'
import Router from 'vue-router'
import Login from '@/views/Login'
import Home from '@/views/Home'
import AdminIndex from '@/views/admin/Index'
import CoachIndex from '@/views/coach/Index'
import UserIndex from '@/views/user/Index'

Vue.use(Router)

// 定义路由表
const routes = [
  { path: '/', redirect: '/login' },
  { path: '/login', component: Login },
  {
    path: '/home',
    component: Home,
    children: [
      { path: 'admin', component: AdminIndex, meta: { role: 1 } },
      { path: 'coach', component: CoachIndex, meta: { role: 2 } },
      { path: 'user', component: UserIndex, meta: { role: 3 } }
    ]
  }
]

const router = new Router({ routes })

// 路由拦截
router.beforeEach((to, from, next) => {
  const token = localStorage.getItem('token')
  // 未登录，拦截到登录页
  if (!token && to.path !== '/login') {
    next('/login')
    return
  }
  // 已登录，校验角色权限
  if (token && to.meta.role) {
    const userInfo = JSON.parse(localStorage.getItem('userInfo'))
    if (userInfo.role !== to.meta.role) {
      // 无权限，跳转到对应角色的首页
      next(`/home/${userInfo.role === 1 ? 'admin' : userInfo.role === 2 ? 'coach' : 'user'}`)
      return
    }
  }
  next()
})

export default router

6 总结与展望
6.1 总结
本文以私人健身房的实际业务需求为导向，完成了一套基于前后端分离架构的私人健身房管理系统的设计与实现。论文首先通过调研分析明确了系统的需求，包括管理员、教练、用户三类角色的功能需求和性能、安全等非功能需求；其次完成了系统的总体架构设计、功能模块设计、数据库设计和界面设计；随后基于 Spring Boot 2 + MyBatis 实现了后端核心接口，基于 Vue2 + ElementUI 实现了前端界面与交互；最后通过功能测试和性能测试验证了系统的可用性和稳定性。
系统的主要成果如下：
采用前后端分离架构，降低了前后端开发的耦合度，提升了系统的可维护性和可扩展性；
实现了精细化的权限管控，基于 RBAC 模型为不同角色分配对应的功能权限，保障系统数据安全；
集成 ECharts 实现数据可视化统计，为健身房运营决策提供了数据支撑；
系统功能贴合私人健身房的实际业务场景，能够替代传统人工管理模式，提升管理效率和用户体验。
6.2 不足与展望
6.2.1 系统不足
数据安全方面，当前版本的用户密码采用明文存储（仅为开发 / 测试版本），生产环境下需优化为加密存储（如 MD5 + 盐值、BCrypt 等）；
功能扩展方面，系统未集成支付接口，用户充值、课程购买仅为模拟功能，无法对接真实的支付渠道；
智能化程度方面，系统缺乏基于用户行为的个性化推荐功能，如根据用户的健身记录推荐课程或教练；
移动端适配方面，系统当前仅适配 PC 端，未开发移动端界面或小程序，用户无法通过移动设备便捷使用系统。
6.2.2 未来展望
优化数据安全机制，将用户密码改为加密存储，增加数据脱敏、接口防刷等安全措施；
集成第三方支付接口（如微信支付、支付宝），实现真实的充值和课程购买功能；
引入大数据分析技术，基于用户的预约记录、课程购买记录等数据，实现个性化课程、教练推荐；
开发移动端适配界面或小程序，支持用户通过手机完成预约、充值、查看课程等操作，提升用户体验；
集成物联网技术，实现健身器材的智能监控，实时采集器材的使用状态、维护需求等数据，进一步提升器材管理的精细化程度。
综上所述，本系统目前已能满足私人健身房的基础管理需求，后续可通过功能扩展和技术优化，进一步提升系统的智能化、便捷性和安全性，为健身行业的数字化转型提供更有力的支撑。
